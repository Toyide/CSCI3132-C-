Encapsulation: imagine the classes Vector3 and Vector2 we had in the previous lab. We declared both with "struct". This means the visibility of all attributes and methods are public by default. Let's see what would happen if we change them to "class" and set the visibility of the attributes to private. Is this going to compile or not? if not how we can fix it?

#include <iostream>
#include <string>

#define LOG(x) std::cout << x << std::endl;

struct Vector {
    void print() {
        // This function does nothing I wish there was a way that I could say user of the class vector cannot call this function
        LOG("This function is an abstract function"); // but what is an abstract function?
    }
};

struct Vector2: public Vector {
    float x, y;

    Vector2 () 
    {
        x = 0;
        y = 0;
    }

    Vector2 (int x, int y) 
    : x(x), y(y) {}

    void print() {
        LOG("This is a vector from Vector2 class");
        std::cout << "(" << x << ", " << y << ")" << std::endl;
    }
};

struct Vector3: public Vector2 {
    float z;

    Vector3(float x, float y, float z) 
    : Vector2(x,y), z(z) {}

    Vector3()
    : Vector2()
    {
        z = 0.0f;
    }
    void print() {
        LOG("This is a vector from Vector3 class");
        std::cout << "(" << x << ", " << y << ", " << z << ")" << std::endl;
    }

};


int main(void) {
    Vector obj1;
    obj1.print();

    Vector2 obj2;
    obj2.print();

    Vector2 obj3(3.2f,2.2f);
    obj3.print();

    Vector3 obj4;
    obj4.print();
    
    Vector3 obj5(5.4f,3.5f,11.1f);
    obj5.print();
   
   return 0;
}
