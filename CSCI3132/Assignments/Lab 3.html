#include <iostream>
#include <cstring>
#define LOG(x)  std::cout << x << std::endl;

int main()
{
    //void* ptr = nullptr; // the simplest pointer  

    /*
    int var = 8;
    void* ptr = &var; // if we change it int it would not change anything  
    */

    /*
    int var = 8;
    int* ptr = &var; // if we change it int it would not change anything  
    // how to de reference to access the data of pointer address
    *ptr = 10;
    */
    /*
    // allocate memory on heap
    char* buffer = new char[8];
    memset(buffer, 0, 8);


    delete[] buffer; // we have to delete it to avoid memory leaks
    */

    char** ptr = &buffer;

    delete[] buffer; // we have to delete it to avoid memory leaks

}
  • How to create an array using pointers
#include <iostream>

#define LOG(x)  std::cout << x << std::endl;

int main()
{
    int example[5]; // now I have array of 5 integers  
    example[0] = 2; // access the first element in the array
    example[4] = 4; // access the last element in the array

    LOG(example[0]);
    LOG(example)

    return 0;
}
#include <iostream>

#define LOG(x)  std::cout << x << std::endl;

int main()
{
    int example[5]; // now I have array of 5 integers  
    int *ptr = example;

    example[2] = 5;
    // This code can be written using pointers
    *(ptr+2) = 6; // first + 2 means we are accessing two elements forward. Then dereference it.

    LOG(example[2]);

    return 0;
}

#include <iostream>

#define LOG(x)  std::cout << x << std::endl;

class NewClass {
public:
    int example[5]; // Stack

    NewClass() {
        for (int i = 0; i < 5; i++)
            example[i] = 2;
    }
};

class OtherClass {
public:
    int* example = new int[5]; // Heap

    OtherClass() {
        for (int i = 0; i < 5; i++)
            example[i] = 2;
    }
};

int main()
{
    NewClass Lab3Variable1;
    LOG(&Lab3Variable1);
    LOG(Lab3Variable1.example);

    LOG(&Lab3Variable2.example);
    LOG(Lab3Variable2.example);

    return 0;
}
