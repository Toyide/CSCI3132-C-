#include <iostream>
#include <string>

#define LOG(x) std::cout << x << std::endl;

struct Vector3 
{
    float x, y, z;

    Vector3(float x, float y, float z) 
    : x(x), y(y), z(z) {}

    Vector3()
    {
        x = 0.0f;
        y = 0.0f;
        z = 0.0f;
    }

    Vector3 Add(const Vector3 & nextObject ) const 
    {
        return Vector3(x + nextObject.x, y + nextObject.y, z + nextObject.z);
    }
    Vector3 operator+(const Vector3 & nextObject ) const 
    {
        return Add(nextObject);
    }

    Vector3 Multiply(const Vector3 & nextObject ) const 
    {
        return Vector3(x * nextObject.x, y * nextObject.y, z * nextObject.z);
    }
    Vector3 operator*(const Vector3 & nextObject ) const 
    {
        return Multiply(nextObject);
    }

    bool operator==(const Vector3& nextObject) const 
    {
        return x == nextObject.x && y == nextObject.y && z == nextObject.z;
    }
    bool operator!=(const Vector3& nextObject) const
    {
        return !(*this == nextObject);
    }
};

std::ostream& operator<<(std::ostream& stream, const Vector3& nextObject)
{
    stream << "(" << nextObject.x << ", " << nextObject.y << ", " << nextObject.z << ")";
    return stream;
}

int main ()
{
    Vector3 position(4.0f, 4.0f, 5.0f); // vector
    Vector3 speed(.05f,1.5f, 1.2f); // vector
    Vector3 powerUp(1.1f, 1.1f, 1.1f);

    Vector3 result1 = position.Add(speed.Multiply(powerUp)); // lets add the vectors together
    Vector3 result2 = position + speed * powerUp; // lets add the vectors together

    Vector3 result3;
    if (result1 == result2) {
        result3 = result1 + result2;
    }

    LOG(result1);
    LOG(result2);
    LOG(result3);

    return 0;
}
